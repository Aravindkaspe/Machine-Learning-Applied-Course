import numpy as np
import matplotlib.pyplot as plt
correlations = data.corr()
correlations
fig = plt.figure()  #open a frame
ax = fig.add_subplot(111)  #create subplpots for each value
cax = ax.matshow(correlations, vmin=-1, vmax=1)  #put the correlations in mat
#start making in graph
fig.colorbar(cax)  #assign a color for each value
ticks = np.arange(0,9,1) #10 columns
ax.set_xticks(ticks)
ax.set_yticks(ticks)
ax.set_xticklabels(names)
ax.set_yticklabels(names)
plt.show()
